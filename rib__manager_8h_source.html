201  void setSweep(int sweepVal)
202  {
203  int ribIndex= luteRibs->selectionIndex();
204  Q_ASSERT(ribIndex<m->seams()+2);
205  if (ribIndex==0) return;
206  if (ribIndex==m->seams()+1){ // 'keel' rib case
207  m->setSeam(ribIndex-2,90.0-0.5*sweepVal);
208  } else {
209  if (ribIndex==1){
210  m->setSeam(0,sweepVal);
211  } else {
212  float seamMid = 0.5*(m->seam(ribIndex-2) + m->seam(ribIndex-1));
213  m->setSeam(ribIndex-2,seamMid-0.5*sweepVal);
214  m->setSeam(ribIndex-1,seamMid+0.5*sweepVal);
215  }
216  }
217  }
