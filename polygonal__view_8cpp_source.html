141 void GLCanvas::easeRotation()
142 {
143  int diffRotX = normalizeDAngle(xRot_target-xRot);
144  int diffRotY = normalizeDAngle(yRot_target-yRot);
145 
146  if (abs(diffRotX)+abs(diffRotY)>16){
147  if (!easingTimer->isActive()) easingTimer->start(10);
148  yRot= normalizeAngle(yRot + EASING_DELAY*diffRotY);
149  xRot= normalizeAngle(xRot + EASING_DELAY*diffRotX);
150  updateGL();
151  } else {
152  easingTimer->stop();
153  }
154 }

175 void GLCanvas::buildDisplaylist()
176 {
177  Part part;
178  Face face;
179  GLfloat colour[4];
180  QColor partColour;
181 
182 
183  glDeleteLists(displayLists,composition->size());
184  displayLists = glGenLists(composition->size());
185 
186  for (int i = 0; i < composition->size() ; ++i){// parts loop
187 
188  glNewList(displayLists+i, GL_COMPILE);
189  part = composition->at(i);
190  partColour=part.getColour();
191  colour[0]=partColour.redF();
192  colour[1]=partColour.greenF();
193  colour[2]=partColour.blueF();
194  colour[3]=partColour.alphaF();
195 
196  for (int j = 0; j < part.size() ; ++j) { // faces loop
197 
198  glMaterialfv(GL_FRONT_AND_BACK, GL_AMBIENT_AND_DIFFUSE, colour);
199  face=part.at(j);
200 
201  if (!face.isConvex()){
202  polygonal::render::tesselate(face,face.normals[0]);
203  } else {
204  glBegin(GL_POLYGON);
205  for (int k = 0; k < face. vertices.size(); ++k) {
206  glNormal3d( face.normals[k].x(),face.normals[k].y(),face.normals[k].z());
207  glVertex3d( face.vertices[k].x(),face.vertices[k].y(),face.vertices[k].z() );
208  }
209  glEnd();
210  }
211  }
212 
213  glEndList();
214  }
215  update();
216 }

290 void GLCanvas::mouseMoveEvent(QMouseEvent *event)
291 {
292  int dx = event->x() - lastPos.x();
293  int dy = event->y() - lastPos.y();
294  lastPos = event->pos();
295 
296  if (event->modifiers() == Qt::ShiftModifier) {
297  yoffset-=MOUSE_PANSTEP*dy*scale;
298  xoffset+=MOUSE_PANSTEP*dx*scale;
299  updateGL();
300  return;
301  }
302 
303  int delta=MOUSE_ROTSTEP;
304  if (easing) delta*=6;
305  setXRotation(xRot + delta * dy);
306  setYRotation(yRot + delta * dx);
307 }
