74 static bool alreadDefined(const QList<QVector3D> &vertexList, const QVector3D &aVertex, int *atIndex)
75 {
76  bool match;
77  for (int i = 0; i < vertexList.size(); ++i) { // iterate over faces
78  match=qFuzzyCompare(aVertex, vertexList[i]);

112 QStringList Polygonal::exportObj(QString path)
113 {
114  QFile objFile(path);
115  QFile mtlFile( QDir(QFileInfo(objFile).path()).filePath(QString("%1.mtl").arg( QFileInfo(objFile).completeBaseName())));
116  QStringList filesCreated;
117  Face face;
118  Part part;
119  int vcount=0;
120  int vcount_marker;
121  bool match;
122  int matchIndex=0;
123  QList<int> faceVertex; // vertex numbers for faces
124  QList<QVector3D> nd; // non-degenerate vertices
125 
126  objFile.open(QIODevice::WriteOnly);
127  mtlFile.open(QIODevice::WriteOnly);
128  QTextStream objStream(&objFile);
129  QTextStream mtlStream(&mtlFile);
130 
131  mtlStream << QString("# %1\n").arg( QFileInfo(mtlFile).fileName() )
132  << "# Material Template Library (Wavefront OBJ-compatible format)\n# Generated "
133  << QDateTime::currentDateTime().toString() << "\n\n";
134 
135  objStream << QString("# %1\n").arg( QFileInfo(objFile).fileName() )
136  <<"# Wavefront OBJ file\n# Generated " << QDateTime::currentDateTime().toString() << "\n\n";
137 
138  objStream << QString("mtllib %1\n").arg(QFileInfo(mtlFile).fileName());
139 
140  for (int partNo = 0; partNo < this->size() ; ++partNo) { // loop over parts
141  part = this->operator[](partNo);
142  vcount_marker=vcount;
143  nd.clear();
144  objStream << "g part"<< partNo<<"\n";
145  objStream << "usemtl part"<< partNo<<"\n";
146  mtlStream << "newmtl part"<< partNo<<"\n";
147  mtlStream << part.mtlColour();
148  for (int i = 0; i < part.size(); ++i) { // loop over faces in part
149  face=part[i];
150  faceVertex.clear();
151  for (int k = 0; k < face.vertices.size(); ++k) {
152  // loop over face vertices, eliminating degeneracies
153  match = alreadDefined(nd, face.vertices[k], &matchIndex);
154  if (match){
155  faceVertex << vcount_marker+matchIndex+1;
156  }else{
157  vcount++;
158  faceVertex<<vcount;
159  nd << face.vertices[k];
160  objStream << "v "
161  << QString::number(face.vertices[k].x()*10,'f',3 ) << " "
162  << QString::number(face.vertices[k].y()*10,'f',3 ) << " "
163  << QString::number(face.vertices[k].z()*10,'f',3 ) << "\n";
164  }
165  }
166 
167  objStream << "f " << faceVertex[0] << "/" << faceVertex[0] << " ";
168 
169  for (int m = 1; m < face.vertices.size(); ++m) {
170  // skip repeated vertex for triads masquerading as quad
171  if (faceVertex[m]!=faceVertex[m-1]) objStream << faceVertex[m] << "/" << faceVertex[m] << " ";
172  }
173  objStream << "\n";
174  }
175  } //parts loop
176 
177  objFile.close();
178  mtlFile.close();
179  return filesCreated << QFileInfo(objFile).fileName() << QFileInfo(mtlFile).fileName();
180 }
